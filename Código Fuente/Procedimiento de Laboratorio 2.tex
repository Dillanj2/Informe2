\section{Procedimiento}

Malla 1

\begin{equation*}
18-820I_1-1000(I_1-I_2)=0
\end{equation*}
\begin{equation*}
18-820I_1-1000I_1+1000I_2=0
\end{equation*}
\begin{equation*}
18-1820I_1+1000I_2=0
\end{equation*}
\begin{equation*}
-1820I_1+1000I_2=-18
\end{equation*}

Malla 2

\begin{equation*}
 -1.2kI_2-2200(I_2-I_3)-1000(I_2-I_1)=0 
\end{equation*}
\begin{equation*}
 -1200I_2-2200I_2+2200I_3-1000I_2+1000I_1=0 
\end{equation*}
\begin{equation*}
 -4400I_2+2200I_3+1000I_1=0
\end{equation*}

Malla 3

\begin{equation*}
 -390I_3-5-2200(I_3-I_2)=0 
\end{equation*}
\begin{equation*}
 -390I_3-5-2200I_3+2200I_2=0
\end{equation*}
\begin{equation*}
 -2590I_3+2200I_2=5
\end{equation*}

Sistema de Ecuaciones con (1),  (2) y (3)

\begin{equation*}
-1820I_1+1000I_2=-18 \\
(1)
\end{equation*}
\begin{equation*}
-4400I_2+2200I_3+1000I_1=0 \\
(2)
\end{equation*}
\begin{equation*}
-2590I_3+2200I_2=5 \\
(3)
\end{equation*}

Resolviendo:

\begin{equation*}
I_1=0.011 A=11.5 mA
\end{equation*} 
\begin{equation*}
I_2=0.00284 A=2.84 mA
\end{equation*}
\begin{equation*}
I_3=0.000488 A=488 uA
\end{equation*}.
